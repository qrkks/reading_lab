# 随机变量及其分布

## 0 1 分布 二项分布

```python
import math
import matplotlib.pyplot as plt

def binomial_term(n, k, p, q):
    """计算二项分布中 P(X=k) 的项"""
    # 组合数 C(n, k)
    comb = math.comb(n, k)
    return comb * (p**k) * (q**(n - k))

# 设定参数
n = 10
p = 0.3
q = 1 - p

# 1. 计算所有项的和
terms = []
for k in range(n + 1):
    term = binomial_term(n, k, p, q)
    terms.append(term)

total_sum = sum(terms)
print(f"参数 n={n}, p={p}")
print(f"各项概率之和 = {total_sum:.10f}")
print(f"(p+q)^n = {(p+q)**n:.10f}")

# 2. 画图
plt.figure(figsize=(8, 5))
plt.bar(range(n + 1), terms, color='skyblue')
plt.title(f'二项分布 B(n={n}, p={p}) 的概率质量函数')
plt.xlabel('成功次数 k')
plt.ylabel('概率 P(X=k)')
plt.xticks(range(n + 1))
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()
```

# 3
# 符号展开
# 定义符号：限定整数、非负
a, b, n, k = sp.symbols('a b n k', integer=True, nonnegative=True)
# 通项：C(n,k) * a^(n−k) * b^k
term = sp.binomial(n, k) * a**(n - k) * b**k
# 对 k 从 0 到 n 求和
binom_expand = sp.summation(term, (k, 0, n))
# 打印求和形式
sp.pprint(binom_expand)

例子：抛硬币 \(n=5\) 次，正面为 “成功”，单次正面概率 \(p=0.5\)
求：恰好出现 2 次正面 的概率

```python
import sympy as sp

n = 5
p = 0.5
q = 1 - p
k = 2

# 计算 P(X=k)
prob = sp.binomial(n, k) * p**k * q**(n - k) # 对，sp.binomial(n, k) 就是专门计算二项式系数 \(\boldsymbol{\dbinom{n}{k}}\)（组合数）。
print(f"P(X={k}) = {prob:.4f}")
```

二项分布

```python
import sympy as sp

n, k = sp.symbols("n k", integer=True, nonnegative=True)
p, g = sp.symbols("p g")
g = 1 - p
expr = sp.binomial(n, k) * p**k * g**(n - k)
print(expr) 

# 代数计算：代入数值：n=400, k=5, p=0.02
val = expr.subs({n: 400, k: 5, p: 0.02}).evalf(2)
print("计算结果：", val)

# 0,1分布
# 设单次概率 p=0.6
p_val = 0.6

# ① 求 P(X=1)：n=1, k=1
res1 = expr.subs({n:1, k:1, p:p_val}).evalf(2) 
# result；
# evalf 是 Evaluate Floating-point 的缩写。
# Evaluate：求值、计算
# Floating-point：浮点数（日常小数）
# 合起来就是计算为浮点数，专门把 SymPy 符号表达式转成小数结果。
print("P(X=1) =", res1)

# ② 求 P(X=0)：n=1, k=0
res0 = expr.subs({n:1, k:0, p:p_val}).evalf(2)
print("P(X=0) =", res0)

print("抛5次硬币，恰好出现2次正面的概率：", expr.subs({n:5, k:2, p:0.5}).evalf(4))
print(expr.subs({n:5, k:2, p:0.5}).evalf(4)*expr.subs({n:5, k:1, p:0.5}).evalf(4))
```
