## 字符串

```lua
print("string")
print(#"string") -- # 返回字符串长度
print([[Hello
    World]])
print("Hello" .. " " .. "World")
print(tostring(10))
print(tonumber("9"))
print(tonumber("9a")) -- 转换失败返回 nil
```

## 函数

```lua
function add(a, b, c) -- 没有冒号
    return a + b, c -- 如果没有返回值，默认返回 nil；可返回多个
end -- 结束 需要 end
print(add(1, 2))
print(add(1, 2, 3)) -- 多了舍去；少了是 nil
```

## Tables

```lua
t = {
    1,
    "a",
    function()
        print(2)
    end,
}
table.insert(t, "insert element") -- 在末尾插入
table.insert(t, 4, "insert ordered element") -- 在第几个元素的前面插入
print(t[1], t[2], t[4], t[5], #t)
t[3]()
pop = table.remove(t, 4)
print(t)
```

```lua
t = {
    a = 1,
    f = function()
        print("I'm a function.")
    end,
}
print(t)
t["f"]()
print(t["a"])
```

## Bool

```lua
a = true
b = false
print(1 > 2)
print(1 < 2)
print(1 >= 2)
print(1 <= 2)
print(1 == 2)
print(1 ~= 2)
print(a and b, a or b, not a)
```

## 分支判断

```lua
if 1 > 10 then
    print("1>10")
elseif 1 < 10 then
    print("1<10")
else
    print("That's impossible")
end

if 0 then
    print("0 is true!")
end
```

## 循环

```lua
for i = 1, 10, 1 do -- for 变量 = 起始值，结束值，步长 do；步长 1 可省略；-1 表示倒序
    print(i)
    if i == 5 then
        print("我准备 break 啦 ")
        break
    end
end
```
