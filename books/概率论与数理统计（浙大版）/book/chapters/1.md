## 全概率公式与贝叶斯公式

```python
import sympy as sp

# 1. 定义符号变量
p1, p2 = sp.symbols('p1 p2')  # P(B1), P(B2)
q1, q2 = sp.symbols('q1 q2')  # P(A|B1), P(A|B2)

# 2. 二元全概率符号表达式
P_A = q1 * p1 + q2 * p2
print("=== 二元全概率 符号表达式 ===")
sp.pprint(P_A)

# 3. 代入本题数值计算
val = P_A.subs({p1: 0.55, p2: 0.45, q1: 0.95, q2: 0.90})
print(f"\n=== 代入数值结果 ===")
print(f"总概率 = {val}")

# 4. 利用划分约束 p1 + p2 = 1 做符号化简
expr_constrain = P_A.subs(p2, 1 - p1)
print("\n=== 带入约束 p2 = 1-p1 表达式 ===")
sp.pprint(expr_constrain)
print("展开后：")
sp.pprint(sp.expand(expr_constrain))

# 5. 一般 n 项求和形式
i, n = sp.symbols('i n', integer=True, positive=True)
p = sp.Function('p')(i)
q = sp.Function('q')(i)
P_A_sum = sp.Sum(p * q, (i, 1, n))
print("\n=== n 项通用全概率求和形式 ===")
sp.pprint(P_A_sum)
```

```python
import sympy as sp

# 把每个条件概率、划分概率都定义为独立符号
p_ab1 = sp.Symbol('P(A|B1)')
p_ab2 = sp.Symbol('P(A|B2)')
p_b1  = sp.Symbol('P(B1)')
p_b2  = sp.Symbol('P(B2)')

# 全概率表达式
pa = p_ab1 * p_b1 + p_ab2 * p_b2
sp.pprint(pa)

# 整体批量代入数值（按整个符号替换）
res = pa.subs({
    p_ab1: 0.95,
    p_ab2: 0.90,
    p_b1:  0.55,
    p_b2:  0.45
})
print("\n计算结果：", res)
```
