# 接口

[接口相当于是方法的类型](../appendices/11.md#接口相当于是方法的类型)

```go
import "fmt"

type Chicken struct {
	Name string
}

func (c Chicken) Speak() string {
	return "咯咯哒"
}

type Duck struct {
	Name string
}

func (d Duck) Speak() string {
	return "呱呱"
}

type Speaker interface {
	Speak() string
}

func MakeSpeak(s Speaker) {
	fmt.Println(s.Speak())
}

chicken := Chicken{Name: "小黄"}

duck := Duck{Name: "唐老鸭"}
fmt.Println(chicken.Speak())
fmt.Println(duck.Speak())
MakeSpeak(chicken)
MakeSpeak(duck)
```
