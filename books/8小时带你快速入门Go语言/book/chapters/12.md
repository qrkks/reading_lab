# 协程和 channel

## 协程

```go
import (
	"fmt"
	"time"
)

func shopping(name string) {
	fmt.Printf("%s 开始购物\n", name)
	time.Sleep(1 * time.Second)
	fmt.Printf("%s 购物结束\n", name)
}
func main() {
	startTime := time.Now()
	fmt.Println(startTime)
	shopping("张三")
	shopping("李四")
	shopping("王五")
	fmt.Println("购买完成", time.Since(startTime))
}

main()
```

```go
import (
	"fmt"
	"sync"
	"time"
)

func shopping(name string, wg *sync.WaitGroup) { // 传 wg 必须传指针
	defer wg.Done() // 减数
	// defer 的价值在于：提前登记清理工作，确保函数从中间退出时也会执行。因此一般放在开头
	// defer 要先执行到并登记；登记后，普通返回和 panic 退出时都会执行，但 os.Exit 等强制终止不会执行。
	fmt.Printf("%s 开始购物\n", name)
	time.Sleep(1 * time.Second)
	fmt.Printf("%s 购物结束\n", name)
}
func main() {
	var wg sync.WaitGroup

	wg.Add(3) // 计数
	startTime := time.Now()
	fmt.Println(startTime)
	go shopping("张三", &wg) // 必须传指针！
	go shopping("李四", &wg)
	go shopping("王五", &wg)
	wg.Wait()
	fmt.Println("购买完成", time.Since(startTime))
}

main()
```

## Channel

```go
import (
	"fmt"
	"sync"
	"time"
)

func pay(
	name string,
	money int,
	moneyChan chan<- int,
	wg *sync.WaitGroup,
) {
	defer wg.Done()

	fmt.Printf("%s 开始购物\n", name)
	time.Sleep(time.Second)
	fmt.Printf("%s 购物结束，消费 %d 元\n", name, money)

	moneyChan <- money
}
func main() {
	var wg sync.WaitGroup
	moneyChan := make(chan int)

	wg.Add(3)

	startTime := time.Now()

	go pay("张三", 2, moneyChan, &wg)
	go pay("李四", 3, moneyChan, &wg)
	go pay("王五", 5, moneyChan, &wg)

	// 等所有付款任务完成后关闭 channel
	go func() {
		wg.Wait()
		close(moneyChan)
	}()

	total := 0

	// 不断接收，直到 channel 被关闭
	for money := range moneyChan {
		total += money
	}

	fmt.Printf(
		"购买完成，总计 %d 元，耗时 %v\n",
		total,
		time.Since(startTime),
	)
}

main()
```
