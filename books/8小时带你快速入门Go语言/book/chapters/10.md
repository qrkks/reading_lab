```go
import "fmt"

type Duck string

func (d Duck) shout() {
	fmt.Println(d, "：呱呱")
}

var don Duck = "Donald"

don.shout()
```

## 类型别名

```go
import "fmt"

type Mouse string
type Cat = string // 类型别名不能绑方法

mouse := Mouse("Jerry")
cat := Cat("Tom")

fmt.Printf("%v,%T\n", mouse, mouse) //v value
fmt.Printf("%v,%T\n", cat, cat)     //T Type
```
