# 结构体

```go
import "fmt"

type User struct {
	username string
	email    string
}

var user = User{
	username: "qrkks",
	email:    "qrkks3@gmail.com", // 由于 go 的分号（;）自动插入机制，末尾必须加逗号，不然就会被插入分号
}

func (u User) Introduce() {
	fmt.Printf("Hello, I am %s, my email is %s\n", u.username, u.email)
}
func New(name, email string) *User {
	return &User{name, email}
}

var user2 = New("new qrkks", "new qrkks3@gmail.com")

fmt.Println(user)
user.Introduce()
fmt.Println(user2)
fmt.Println(*user2)
```

```go
import "fmt"

type User struct {
	Username string `json:"username, omitempty"` // 字段名大写可以跨包
	// omitempty 抛弃空值
	// 没有 tag 时返回字段名
	Email string `json:"-"`
	// Email string `json:"-"`
	// 表示 encoding/json 完全忽略这个字段：
	// 转成 JSON 时不输出
	// 从 JSON 解析回来时也不接收
	// 字段仍然存在于 Go 结构体中，正常 Go 代码可以使用
	Group `json:"group"` // 匿名嵌入，不只是简写；此时 Group 的属性都提到外层 User 上，不建议使用，此处仅为示例
	// 仍然是匿名嵌入字段，但在转 JSON 时，会把它作为名为 group 的嵌套对象。
}
type Group struct {
	Name string
}

func (u *User) SetName(name string) {
	u.Username = name
}

admin := Group{"管理员"}
fmt.Println(admin)

user := User{
	Username: "qrkks", // 尽量使用字段名实例化，不使用位置，可以在修改的时候更
	Email:    "qrkks3@gmail.com",
	Group:    admin, // 换行布局的时候最后一个也要补逗号，不然会被自动补分号造成报错
}

fmt.Println(user)
user.SetName("new name")
fmt.Println(user)
```
