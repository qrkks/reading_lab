# 线程安全

## 线程不安全

```go
import "fmt"
import "time"
import "sync"

var sum int

func add() {
	defer wg.Done()
	for i := 0; i < 10000000; i++ {
		sum++
	}
}

var wg sync.WaitGroup // 声明，不是赋值

start := time.Now()
fmt.Println(start)

// 必须在启动 goroutine 前设置计数。不能放在 goroutine 中。
wg.Add(2)

go add()
go add()

wg.Wait()

fmt.Println(sum)
fmt.Print(time.Since(start))
```

## 加锁

```go
import "fmt"
import "time"
import "sync"

var sum int

func add() {
	defer wg.Done()
	for i := 0; i < 100000; i++ {
		mu.Lock()
		sum++
		mu.Unlock()
	}
}

var wg sync.WaitGroup // 声明，不是赋值
var mu sync.Mutex

start := time.Now()
fmt.Println(start)

// 必须在启动 goroutine 前设置计数。不能放在 goroutine 中。
wg.Add(2)

go add()
go add()

wg.Wait()

fmt.Println(sum)
fmt.Print(time.Since(start))
```

[只有一个协程相当于还是同步的？](../appendices/13.md#只有一个协程相当于还是同步的)

```go
import (
	"fmt"
	"time"
)

const count = 100_000_000

func add() int {
	sum := 0

	for i := 0; i < count; i++ {
		sum++
	}

	return sum
}
func main() {
	// --------------------
	// 同步版本
	// --------------------

	startSync := time.Now()

	sum1 := add()
	sum2 := add()

	syncResult := sum1 + sum2
	syncDuration := time.Since(startSync)

	fmt.Println("同步结果：", syncResult)
	fmt.Println("同步耗时：", syncDuration)

	// --------------------
	// 并发版本
	// --------------------

	startConcurrent := time.Now()

	resultCh := make(chan int, 2)

	go func() {
		resultCh <- add()
	}()

	go func() {
		resultCh <- add()
	}()

	result1 := <-resultCh
	result2 := <-resultCh

	concurrentResult := result1 + result2
	concurrentDuration := time.Since(startConcurrent)

	fmt.Println("并发结果：", concurrentResult)
	fmt.Println("并发耗时：", concurrentDuration)

	// --------------------
	// 比较
	// --------------------

	if concurrentDuration < syncDuration {
		fmt.Printf(
			"并发版本更快，大约快 %.2f 倍\n",
			float64(syncDuration)/float64(concurrentDuration),
		)
	} else {
		fmt.Printf(
			"同步版本更快，大约快 %.2f 倍\n",
			float64(concurrentDuration)/float64(syncDuration),
		)
	}
}

// 每个任务独立计算自己的结果。

main()
```
