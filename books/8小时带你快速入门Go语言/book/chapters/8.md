# init 函数和 defer 函数

```go
import "fmt"

func init() {
	fmt.Println("init-1")
}
func init() {
	fmt.Println("init-2")
}
func init() {
	fmt.Println("init-3")
}

fmt.Println("main")
```

```go
import "fmt"

func f() {
	defer func() {
		fmt.Println("defer")
	}()
	defer fmt.Println("defer direct") // 如果有多个 defer 会后写的先执行，后进先出
	func() {
		fmt.Println("normal")
	}()
}

f()
```
